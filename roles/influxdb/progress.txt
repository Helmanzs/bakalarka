lxc launch ubuntu:24.04 influxdb --vm --profile default --profile influxdb-net -c limits.memory=4GiB
lxc config device add influxdb shared disk source=~/data path=/data
lxc file push ~/playbook/create/influxdb/process_data.sh influxdb/opt/
lxc file push ~/playbook/create/influxdb/send_data.sh influxdb/opt/
lxc exec influxdb -- bash
apt update && sudo apt upgrade -y
apt install python3 python3-pip -y
mkdir -p /root/.influxdb

curl -sSL https://dl.influxdata.com/influxdb/releases/influxdb3-core-3.0.3_linux_amd64.tar.gz -o /root/.influxdb/influxdb3-core.tar.gz

tar -xf /root/.influxdb/influxdb3-core.tar.gz --strip-components=1 -C /root/.influxdb
rm /root/.influxdb/influxdb3-core.tar.gz
ln -s /root/.influxdb/ /influxdb
mkdir -p /influxdb/data
echo 'export PATH="$PATH:/root/.influxdb"' >> ~/.bashrc
source ~/.bashrc

tee /etc/systemd/system/influxdb3.service > /dev/null <<EOF
[Unit]
Description=InfluxDB 3.x Service
After=network.target

[Service]
User=root
ExecStart=/root/.influxdb/influxdb3 serve \
  --node-id=node0 \
  --http-bind=0.0.0.0:8181 \
  --object-store=file \
  --data-dir /root/.influxdb/data
Restart=always

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable influxdb3
systemctl start influxdb3
